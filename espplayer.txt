-- PlayerLocationScript

-- İsteğe bağlı: Çizgi rengini ve kalınlığını buradan değiştirebilirsiniz.
local lineColor = Color3.new(1, 1, 1) -- Beyaz renk
local lineWidth = 2 -- Çizgi kalınlığı

-- Çizgiyi oluşturan fonksiyon
local function createLine(player)
    local line = Instance.new("Frame")
    line.Name = "PlayerLine"
    line.BackgroundColor3 = lineColor
    line.BorderSizePixel = 0
    line.Size = UDim2.new(0, 50, 0, 50) -- Çizgi boyutunu istediğiniz gibi ayarlayabilirsiniz
    line.Parent = player.PlayerGui -- Çizgiyi PlayerGui'ye ekleyin

    -- Çizgiyi güncelleme fonksiyonu
    local function updateLine()
        if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
            local position = workspace.CurrentCamera:WorldToViewportPoint(player.Character.HumanoidRootPart.Position)
            line.Position = UDim2.new(0, position.X, 0, position.Y)
        else
            line:Destroy()
        end
    end

    -- Oyuncunun pozisyonunu güncelleme
    player.CharacterAdded:Connect(function()
        updateLine()
    end)

    player.CharacterRemoving:Connect(function()
        line:Destroy()
    end)

    -- Her frame'de güncelleme
    game:GetService("RunService").RenderStepped:Connect(function()
        updateLine()
    end)
end

-- Tüm oyuncular için çizgileri oluştur
for _, player in pairs(game.Players:GetPlayers()) do
    createLine(player)
end

-- Oyuncu eklediğinde yeni çizgi oluştur
game.Players.PlayerAdded:Connect(function(player)
    createLine(player)
end)
